export function useAvatarsList(args:
  | { resourceType: "body"; query: BodyListQuery }
  | { resourceType: "face"; query: FaceListQuery }
) {
  return useQuery({
    queryKey: ["avatars", args.resourceType, "list", args.query],
    queryFn: () => args.resourceType === "body"
      ? listBodies(args.query as BodyListQuery)
      : listFaces(args.query as FaceListQuery),
  })
}

### Task G3.4: usePublishAvatar / useRetireAvatar (TDD)

```ts
export function usePublishAvatar(resourceType: AvatarResourceType) {
  const qc = useQueryClient()
  return useMutation({
    mutationFn: (id: number) =>
      resourceType === "body" ? publishBody(id) : publishFace(id),
    onSuccess: () => {
      qc.invalidateQueries({ queryKey: ["avatars", resourceType] })
      mutationSuccessToast(resourceType === "body" ? "Body 게시 완료" : "Face 게시 완료")
    },
    onError: mutationErrorToast,
  })
}

export function useRetireAvatar(resourceType: AvatarResourceType) {
  const qc = useQueryClient()
  return useMutation({
    mutationFn: (vars: { id: number; body: RetireRequest }) =>
      resourceType === "body" ? retireBody(vars.id, vars.body) : retireFace(vars.id, vars.body),
    onSuccess: () => {
      qc.invalidateQueries({ queryKey: ["avatars", resourceType] })
      mutationSuccessToast(resourceType === "body" ? "Body 회수 완료" : "Face 회수 완료")
    },
    onError: mutationErrorToast,
  })
}
```

Tests: 6 (resourceType별 toast / invalidate prefix / AVT-005 error 미invalidate).

**Commit**: `feat(avatars): add usePublishAvatar + useRetireAvatar mutations (G3.4)`
