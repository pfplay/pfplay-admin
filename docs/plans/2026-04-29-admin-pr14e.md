```ts
export const TERMINAL_STATUSES = new Set<ReportStatus>(["RESOLVED", "DISMISSED"])
export const TRANSITION_MATRIX: Record<ReportStatus, ReportStatus[]> = {
  PENDING: ["REVIEWING", "DISMISSED"],
  REVIEWING: ["RESOLVED", "DISMISSED", "PENDING"],
  RESOLVED: [],
  DISMISSED: [],
}

// PATCH body schema — terminal 진입 시 resolutionNote 필수 refine
export const ReportStatusUpdateSchema = z.object({
  status: z.enum(["PENDING", "REVIEWING", "RESOLVED", "DISMISSED"]),
  resolutionNote: z.string().max(2000).optional().nullable(),
}).refine(
  (v) => !TERMINAL_STATUSES.has(v.status as ReportStatus) || (v.resolutionNote?.trim().length ?? 0) >= 1,
  { message: "처리 완료/기각 시 처리 메모는 필수입니다", path: ["resolutionNote"] },
)
export type ReportStatusUpdateRequest = z.infer<typeof ReportStatusUpdateSchema>

export function canTransition(from: ReportStatus, to: ReportStatus): boolean {
  return TRANSITION_MATRIX[from].includes(to)
}
```

- 5 transition 모두 허용
- 위반 (RESOLVED→anything) reject
- terminal target + empty resolutionNote refine fail
- terminal target + filled note OK
- non-terminal + empty note OK
- canTransition helper 6 case

```ts
import { http } from "@/shared/api/http"
import type { Page } from "@/shared/api/page"
import { unwrap } from "@/shared/api/page"
import { serializeQuery } from "@/shared/lib/url-state"
import type { AdminReportSummary, AdminReportDetail } from "@/entities/report"
import type { ReportsListQuery } from "../model/filter-schema"
import type { ReportStatusUpdateRequest } from "../model/transition-schema"
import type { ApiCommonResponse } from "@/shared/api/page"

export async function listReports(query: ReportsListQuery): Promise<Page<AdminReportSummary>> {
  // status[]/category[] multi serialization — URLSearchParams.append 패턴
  const params = new URLSearchParams()
  if (query.status) query.status.forEach((s) => params.append("status", s))
  if (query.category) query.category.forEach((c) => params.append("category", c))
  if (query.createdFrom) params.set("created_from", query.createdFrom)
  if (query.createdTo) params.set("created_to", query.createdTo)
  params.set("page", String(query.page))
  params.set("size", String(query.size))
  params.set("sort", query.sort)
  const qs = params.toString()
  return unwrap(
    await http<ApiCommonResponse<Page<AdminReportSummary>>>(`/api/v1/admin/reports?${qs}`),
  )
}

export async function getReportDetail(reportId: number): Promise<AdminReportDetail> {
  return unwrap(
    await http<ApiCommonResponse<AdminReportDetail>>(`/api/v1/admin/reports/${reportId}`),
  )
}

export async function updateReportStatus(
  reportId: number, body: ReportStatusUpdateRequest,
): Promise<AdminReportDetail> {
  return unwrap(
    await http<ApiCommonResponse<AdminReportDetail>>(`/api/v1/admin/reports/${reportId}`, {
      method: "PATCH", body,
    }),
  )
}
```

### Task G3.4: useUpdateReportStatus (TDD)

**Files:** `use-update-report-status.ts` + test

핵심 차이점 (14d 대비):

```ts
export function useUpdateReportStatus() {
  const qc = useQueryClient()
  return useMutation({
    mutationFn: (vars: { reportId: number; body: ReportStatusUpdateRequest }) =>
      updateReportStatus(vars.reportId, vars.body),
    onSuccess: (newDetail, vars) => {
      // PATCH 응답이 detail shape → 즉시 cache 갱신 (refetch 회피)
      qc.setQueryData(["reports", "detail", vars.reportId], newDetail)
      // list cache는 stale 마킹 (status badge / reviewedBy / resolvedAt 변경 반영)
      qc.invalidateQueries({ queryKey: ["reports", "list"] })
      // toast 분기 — target status에 따라
      const labels: Record<string, string> = {
        PENDING: "보류 처리", REVIEWING: "검토 시작",
        RESOLVED: "처리 완료", DISMISSED: "기각 처리",
      }
      mutationSuccessToast(labels[newDetail.status] ?? "상태 변경 완료")
    },
    onError: mutationErrorToast,
  })
}
```

테스트:
- success: setQueryData(detail) + invalidate(list) + toast.success 메시지 정확
- 4 transition 메시지 분기 (PENDING/REVIEWING/RESOLVED/DISMISSED)
- 400 RPT-002 → invalidate/setQueryData 미호출 + mutationErrorToast (ApiError 그대로)

**Commit**: `feat(reports): add useUpdateReportStatus (setQueryData + invalidate split + toast 분기) (G3.4)`
